# Setup instructions for Windows PowerShell
# Do not forget to type the command : cd django_project

# 1. Create virtual environment
python -m venv venv

# 2. Activate virtual environment
.\venv\Scripts\Activate.ps1

# 3. Upgrade pip and install requirements
pip install --upgrade pip
pip install -r requirements.txt

# 4. Run migrations (IMPORTANT for CKEditor)
python manage.py makemigrations
python manage.py migrate

# 5. Create sample data
python manage.py create_sample_data

# 6. Run the server
python manage.py runserver

# ⚠️ Troubleshooting:
# If you have issues, try: python manage.py flush
# ⚠️ Important: Do NOT push the venv/ folder to GitHub
# It is excluded via .gitignore
